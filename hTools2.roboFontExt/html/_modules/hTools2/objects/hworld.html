[docs]    def projects(self):
        '''Returns a list of all projects contained in the root fonts folder.'''
        allFiles = os.listdir(self.settings.root)
        projects = []
        for n in allFiles:
            # hTools2 convention: project folders start with _
            if n[:1] == "_":
                projects.append(n[1:])
        return projects
